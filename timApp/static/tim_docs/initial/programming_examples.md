         * haeLuku(new JTextField("234.233"), 44) ~~~ 234.23;
         * haeLuku(new JTextField("foobar1"), 44) ~~~ 44.00;
         * haeLuku(new JTextField("1foobar"), 44) ~~~ 1.00;
